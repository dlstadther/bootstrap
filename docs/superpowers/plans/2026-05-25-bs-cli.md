# bs CLI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the `bs` CLI as a compiled Go binary with Cobra, replacing the bash-dispatcher approach in bs-cz1 with a proper, testable CLI.

**Architecture:** `bs/` at repo root is a self-contained Go module. `cmd/` holds thin Cobra wrappers that parse flags and delegate. `internal/` holds all business logic and tests. Shell commands go through an injectable `Executor` interface so unit tests never invoke real processes.

**Tech Stack:** Go 1.22, github.com/spf13/cobra v1.8.0

Create `bs/internal/exec/exec.go`:

```go
package exec

import (
	"bytes"
	gexec "os/exec"
	"os"
	"strings"
)

// Executor runs shell commands. Real uses os/exec; Fake is used in tests.
type Executor interface {
	Run(name string, args ...string) (string, error)
	Stream(name string, args ...string) error
}

// Real executes commands via os/exec.
type Real struct{}

func (Real) Run(name string, args ...string) (string, error) {
	cmd := gexec.Command(name, args...)
	var buf bytes.Buffer
	cmd.Stdout = &buf
	cmd.Stderr = &buf
	err := cmd.Run()
	return strings.TrimSpace(buf.String()), err
}

func (Real) Stream(name string, args ...string) error {
	cmd := gexec.Command(name, args...)
	cmd.Stdout = os.Stdout
	cmd.Stderr = os.Stderr
	return cmd.Run()
}
```

- [ ] **Step 2: Create fake.go**

Create `bs/internal/exec/fake.go`:
```go
package exec

import "slices"

// Response is a canned reply for a specific command + args pair.
type Response struct {
	Name string
	Args []string
	Out  string
	Err  error
}

// Call records one invocation of Run or Stream.
type Call struct {
	Name   string
	Args   []string
	Stream bool
}

// Fake is a test double for Executor. Pre-load Responses; inspect Calls after.
type Fake struct {
	Responses []Response
	Calls     []Call
}

func NewFake() *Fake { return &Fake{} }

func (f *Fake) Run(name string, args ...string) (string, error) {
	f.Calls = append(f.Calls, Call{Name: name, Args: args})
	for _, r := range f.Responses {
		if r.Name == name && slices.Equal(r.Args, args) {
			return r.Out, r.Err
		}
	}
	return "", nil
}

func (f *Fake) Stream(name string, args ...string) error {
	f.Calls = append(f.Calls, Call{Name: name, Args: args, Stream: true})
	for _, r := range f.Responses {
		if r.Name == name && slices.Equal(r.Args, args) {
			return r.Err
		}
	}
	return nil
}
```

```go
package git

import (
	"os"
	"strings"

	bsexec "github.com/dlstadther/bootstrap/bs/internal/exec"
	"github.com/dlstadther/bootstrap/bs/internal/version"
)

// RepoPath returns the bootstrap repo path.
// Precedence: $BOOTSTRAP_REPO env var > compiled-in version.RepoPath.
func RepoPath() string {
	if p := os.Getenv("BOOTSTRAP_REPO"); p != "" {
		return p
	}
	return version.RepoPath
}

// CurrentHash returns the short git hash of HEAD in the repo.
func CurrentHash(ex bsexec.Executor) (string, error) {
	return ex.Run("git", "-C", RepoPath(), "rev-parse", "--short", "HEAD")
}

// IsDirty returns true if the repo has uncommitted changes.
func IsDirty(ex bsexec.Executor) (bool, error) {
	out, err := ex.Run("git", "-C", RepoPath(), "status", "--porcelain")
	if err != nil {
		return false, err
	}
	return strings.TrimSpace(out) != "", nil
}
```

```go
package cmd

import (
	"fmt"

	"github.com/spf13/cobra"
	bsexec "github.com/dlstadther/bootstrap/bs/internal/exec"
	"github.com/dlstadther/bootstrap/bs/internal/git"
	"github.com/dlstadther/bootstrap/bs/internal/version"
)

func init() {
	rootCmd.AddCommand(versionCmd)
}

var versionCmd = &cobra.Command{
	Use:   "version",
	Short: "Show compiled and current repo version",
	RunE: func(cmd *cobra.Command, args []string) error {
		return runVersion(bsexec.Real{})
	},
}

func runVersion(ex bsexec.Executor) error {
	fmt.Printf("compiled: %-10s (%s)\n", version.CommitHash, version.BuildTime)

	repoHash, err := git.CurrentHash(ex)
	if err != nil {
		fmt.Printf("repo:     (unavailable — %v)\n", err)
		return nil
	}

	dirty, err := git.IsDirty(ex)
	if err != nil {
		fmt.Printf("repo:     %-10s (status unavailable)\n", repoHash)
		return nil
	}

	annotation := "clean"
	if version.CommitHash != repoHash {
		annotation = "run 'make install' to update"
	} else if dirty {
		annotation = "uncommitted changes"
	}

	fmt.Printf("repo:     %-10s (%s)\n", repoHash, annotation)
	return nil
}
```

```go
package brew

import (
	"fmt"

	bsexec "github.com/dlstadther/bootstrap/bs/internal/exec"
)

const tmpBrewfile = "/tmp/.Brewfile.current"

// Sync shows drift between the repo Brewfile and the live machine state.
func Sync(ex bsexec.Executor, repoPath string) error {
	brewfilePath := repoPath + "/dotfiles/.Brewfile"
	if _, err := ex.Run("brew", "bundle", "dump", "--force", "--file="+tmpBrewfile); err != nil {
		return fmt.Errorf("brew dump: %w", err)
	}
	// diff exits 1 when files differ — that's expected, not an error
	ex.Run("diff", brewfilePath, tmpBrewfile) //nolint:errcheck
	return nil
}

// Dump writes the live brew state back to the repo Brewfile.
func Dump(ex bsexec.Executor, repoPath string) error {
	brewfilePath := repoPath + "/dotfiles/.Brewfile"
	if err := ex.Stream("brew", "bundle", "dump", "--force", "--file="+brewfilePath); err != nil {
		return fmt.Errorf("brew dump: %w", err)
	}
	return nil
}

// Install updates brew and installs packages from the global Brewfile.
func Install(ex bsexec.Executor) error {
	if err := ex.Stream("brew", "update"); err != nil {
		return fmt.Errorf("brew update: %w", err)
	}
	return ex.Stream("brew", "bundle", "install", "--global", "--verbose")
}
```

```go
package tmux

import (
	"fmt"
	"path/filepath"
	"slices"
	"strings"

	bsexec "github.com/dlstadther/bootstrap/bs/internal/exec"
)

var allowedAgents = []string{"claude", "codex", "gemini", "opencode", "pi"}

// WorkspaceConfig holds parameters for a new tmux workspace window.
type WorkspaceConfig struct {
	Name  string // window name override; defaults to basename(CWD)
	CWD   string // required; working directory for all panes
	Agent string // agent to stage in left pane; defaults to "claude"
	Dev   bool   // reserved for future dev-layout variant
}

// Validate returns an error if the config is invalid.
func (c WorkspaceConfig) Validate() error {
	if c.CWD == "" {
		return fmt.Errorf("--cwd is required")
	}
	if c.Agent != "" && !slices.Contains(allowedAgents, c.Agent) {
		return fmt.Errorf("invalid agent %q; allowed: %s", c.Agent, strings.Join(allowedAgents, ", "))
	}
	return nil
}

func (c WorkspaceConfig) agent() string {
	if c.Agent == "" {
		return "claude"
	}
	return c.Agent
}

// Add creates a new tmux workspace: new session if none exists, new window otherwise.
// Replicates the pane layout from dotfiles/.config/tmux/templates/workspace.sh.
func Add(ex bsexec.Executor, cfg WorkspaceConfig) error {
	session := filepath.Base(cfg.CWD)
	dir := cfg.CWD

	_, sessionErr := ex.Run("tmux", "has-session", "-t", session)
	if sessionErr != nil {
		window := cfg.Name
		if window == "" {
			window = session
		}
		if _, err := ex.Run("tmux", "new-session", "-d", "-s", session, "-n", window, "-c", dir); err != nil {
			return fmt.Errorf("new-session: %w", err)
		}
		if err := createPanes(ex, session, window, dir, cfg.agent()); err != nil {
			return err
		}
		ex.Run("tmux", "switch-client", "-t", session) //nolint:errcheck
		return nil
	}

	window := resolveWindowName(ex, session, cfg)
	if _, err := ex.Run("tmux", "new-window", "-t", session, "-n", window, "-c", dir); err != nil {
		return fmt.Errorf("new-window: %w", err)
	}
	if err := createPanes(ex, session, window, dir, cfg.agent()); err != nil {
		return err
	}
	ex.Run("tmux", "select-window", "-t", session+":"+window) //nolint:errcheck
	return nil
}

func resolveWindowName(ex bsexec.Executor, session string, cfg WorkspaceConfig) string {
	if cfg.Name != "" {
		return cfg.Name
	}
	base := filepath.Base(cfg.CWD)
	out, _ := ex.Run("tmux", "list-windows", "-t", session, "-F", "#{window_name}")
	existing := strings.Split(out, "\n")
	candidate := base
	for n := 2; slices.Contains(existing, candidate); n++ {
		candidate = fmt.Sprintf("%s-%d", base, n)
	}
	return candidate
}

func createPanes(ex bsexec.Executor, session, window, dir, agent string) error {
	target := session + ":" + window

	// Right column (40%)
	if _, err := ex.Run("tmux", "split-window", "-h", "-p", "40", "-t", target, "-c", dir); err != nil {
		return err
	}
	// Split right into top + bottom; bottom runs git/bd
	if _, err := ex.Run("tmux", "split-window", "-v", "-t", target, "-c", dir); err != nil {
		return err
	}
	ex.Run("tmux", "send-keys", "-t", target, "git pull origin main && ls -al && bd ready", "Enter") //nolint:errcheck

	// Split again; bottom runs lazygit
	if _, err := ex.Run("tmux", "split-window", "-v", "-t", target, "-c", dir); err != nil {
		return err
	}
	ex.Run("tmux", "send-keys", "-t", target, "lazygit", "Enter") //nolint:errcheck

	// Left pane: stage agent — no Enter, user stays in control
	ex.Run("tmux", "select-pane", "-t", target+".0")      //nolint:errcheck
	ex.Run("tmux", "send-keys", "-t", target+".0", agent) //nolint:errcheck
	return nil
}
```

```go
import (

	bsexec "github.com/dlstadther/bootstrap/bs/internal/exec"
	"github.com/dlstadther/bootstrap/bs/internal/tmux"
)

	name  string
	cwd   string
	agent string
	dev   bool
}

var addCmd = &cobra.Command{
	Use:   "add",
	Short: "Create a tmux workspace session or window",
	RunE: func(cmd *cobra.Command, args []string) error {
		cfg := tmux.WorkspaceConfig{
			Name:  addFlags.name,
			CWD:   addFlags.cwd,
			Agent: addFlags.agent,
			Dev:   addFlags.dev,
		}
		if err := cfg.Validate(); err != nil {
			return err
		}
		return tmux.Add(bsexec.Real{}, cfg)
	},
}

	addCmd.Flags().StringVar(&addFlags.agent, "agent", "", "Agent to stage in left pane (default: claude)")
	addCmd.Flags().BoolVar(&addFlags.dev, "dev", false, "Dev workspace layout (reserved for future use)")
```
